<script>

    let exercises = JSON.parse(localStorage.getItem('myWorkoutsV2')) || [];

    let timerInterval;

 

    function renderExercises() {

        const list = document.getElementById('exerciseList');

        list.innerHTML = '';

        exercises.forEach((ex, index) => {

            list.innerHTML += `

                <div class="exercise-card">

                    ${ex.image ? <img src="${ex.image}" class="exercise-img"> : '<div class="exercise-img" style="display:flex;align-items:center;justify-content:center;color:#94a3b8">Sin foto</div>'}

                    <div class="exercise-info">

                        <button class="delete-btn" onclick="deleteExercise(${index})">Eliminar</button>

                        <h3>${ex.name}</h3>

                        <div class="stats">

                            <span class="tag">📊 ${ex.reps}</span>

                            <span class="tag">⚖️ ${ex.weight}</span>

                        </div>

                    </div>

                </div>

            `;

        });

    }

 

    function addExercise() {

        const name = document.getElementById('name').value;

        const reps = document.getElementById('reps').value;

        const weight = document.getElementById('weight').value;

        const photoFile = document.getElementById('photo').files[0];

 

        if (!name) return alert("Ponle un nombre al menos");

 

        if (photoFile) {

            const reader = new FileReader();

            reader.onloadend = () => saveData(name, reps, weight, reader.result);

            reader.readAsDataURL(photoFile);

        } else {

            saveData(name, reps, weight, null);

        }

    }

 

    function saveData(name, reps, weight, image) {

        exercises.push({ name, reps, weight, image });

        localStorage.setItem('myWorkoutsV2', JSON.stringify(exercises));

        renderExercises();

        // Limpiar campos

        ['name', 'reps', 'weight', 'photo'].forEach(id => document.getElementById(id).value = '');

    }

 

    function deleteExercise(index) {

        if(confirm("¿Borrar ejercicio?")) {

            exercises.splice(index, 1);

            localStorage.setItem('myWorkoutsV2', JSON.stringify(exercises));

            renderExercises();

        }

    }

 

    // --- Lógica del Temporizador ---

    let timeLeft = 0;

        clearInterval(timerInterval);

        timeLeft = seconds;
